//------------------------START OF MAIN()--------------------------------------

{

	printf("(a) 8%%5: %d",8%5);
	printf("\n(b) -8%%5: %d",-8%5);

	printf("\n(d) -8%%-5: %d",-8%-5);
	
	printf("\n++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++\n");
	return 0;
}
