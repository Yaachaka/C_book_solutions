@@@@ Resultant of the given expression will be int (of size 4 bytes).

PROGRAM: (FLAGS: -std=c99): knkcch07e09.c
#include<stdio.h>
//------------------------START OF MAIN()--------------------------------------
int main(void)
{
	printf("++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++\n");
	
	int i,j;
	
	printf("Size of i/j+\'a\' is %d bytes.",sizeof(i/j+'a'));
	
	printf("\n++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++\n");
	return 0;
}
//-------------------------END OF MAIN()---------------------------------------
