if i has the value 17? What does it print if i has the value -17?
	printf("%d\n",i >= 0 ? i : -i);
==========================================================================
SOLUTION:
-- On both cases (i=17 and i=-17), the output will be 17.
-- Check PROGRAM OUTPUT.
==========================================================================
PROGRAM: (FLAGS: -std=c99): knkcch05e07.c
#include<stdio.h>
//------------------------START OF MAIN()--------------------------------------
int main(void)
{
	printf("++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++\n");
	
	int i=17;
	printf("i=17: %d",i >= 0 ? i : -i);
	i=-17;
	printf("\ni=-17: %d",i >= 0 ? i : -i);
	
	printf("\n++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++\n");
	return 0;
}
//-------------------------END OF MAIN()---------------------------------------
//---------------------------------------------------------------------------

++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
i=17: 17
i=-17: 17
