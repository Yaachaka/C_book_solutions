void censor(char a[],int n)
{
	int i=0;

	while(a[i+2]!='\0' && i<n-2)
	{
		if(tolower(a[i])=='f' && tolower(a[i+1])=='o' && tolower(a[i+2])=='o')
		{
			a[i++]='x';
			a[i++]='x';
			a[i++]='x';
		}
		else
			i++;
	}
}
